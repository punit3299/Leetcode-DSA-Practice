class Solution {
public:
    bool validMountainArray(vector<int>& a) {
        
        int n=a.size();
        bool result=false;
        
        if(n>=3){
            int i;
           for(i=0;i<n-1;i++){
               if(a[i]<a[i+1]){
                   result=true;
               }
               else{
                   break;
               }
           }
            if(i==n-1){
                result=false;
            }  
            if(result){
                for(;i<n-1;i++){
                    if(a[i]>a[i+1]){
                        result=true;
                    }
                    else{
                        result=false;
                        break;
                    }
                }
            }
        }
        
        return result;
    }
};
